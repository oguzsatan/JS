    function init() {
      navigator.getBattery().then(b => {
        b.onlevelchange = e =>
          level.innerText =
          'Batarya Durumu: ' + 100 * b.level + '%'

        b.onchargingchange = e => charge.innerText = 'Şarjda Mı?: ' + (b.charging ? "Şarjda" : "Şarjda Değil")


      })
      ononline = e => online.innerText = "Online"
      onoffline = e => online.innerText = "Online Değil"
      onkeydown = e => keyboard.innerText = e.key
      onkeyup = e => keyboard.innerText = e.key
    }
    
    sample.innerText = init + '\n';
    init();
